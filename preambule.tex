\renewcommand{\section}[2][]{
	\ifstrempty{#1}%
	{\oldsection{#2}}%
	{\oldsection[#1]{#2}}	
	\addtostream{CONTENT}{%
			\protect\noindent%
			\protect\stepcounter{SEC}%
			\protect\setcounter{SUB}{0}%
			\protect\hspace{3ex}%
			\protect\theSEC ~ #1 \protect\par%
			}
}

\renewcommand{\subsection}[1]{
	\oldsubsection{#1}
	\addtostream{CONTENT}{%
			\protect\noindent%
			\protect\stepcounter{SUB}%
			\protect\hspace{6ex}%
			\protect\theSEC{}.\protect\theSUB ~ #1 \protect\par%
			}
}
