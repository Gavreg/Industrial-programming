\begin{frame}[fragile]\frametitle{git}
	
	\begin{alertblock}{git ---}
		это распределенная система контроля версий, разработанная Линусом Торвальдсом в 2005 году для управления исходным кодом ядра Linux.
	\end{alertblock}
	
	Предназначение:
	
	\begin{itemize}
		\item Отслеживание изменений в файлах
		\item Совместная работа над проектами
		\item Возможность возврата к предыдущим версиям
	\end{itemize}
	
	Установка Windows (через актуальную(!!) версию Windows Installer):
	
	\begin{VerbatimCmd}[]
				
		> !color{orange}winget install Git.Git		
			
	\end{VerbatimCmd}
	
	Установка Linux (через apt):
	
	\begin{VerbatimCmd}[commandchars=!\{\},gobble=2]
				
		> !color{orange}sudo apt update
		> !color{orange}sudo apt install git	
					
	\end{VerbatimCmd}
	
	Установка Mac (через Homebrew):
	
	\begin{VerbatimCmd}[commandchars=!\{\},gobble=2]
		
		> !color{orange}brew install git	
		
	\end{VerbatimCmd}
	
	
\end{frame}

\begin{frame}[fragile]\frametitle{Работа через командную строку}
	
	\textbf{Проверка установки}
	\begin{VerbatimCmd}
		
		> !color{orange}git --version
		git version 2.48.1.windows.1
		
	\end{VerbatimCmd}
	
	\medskip
	
	
	\textbf{Основной синтаксис}

		> !color{orange}git команда -h !color{gray} #Сокращённая, в терминале
				
	\end{VerbatimCmd}

\begin{frame}[fragile]\frametitle{Настройка git}
	
	\begin{alertblock}{\Verb|git config|}
	  позволяет задавать настройки git на уровне системы или отдельного репозитория.
		
	\end{alertblock}
	
	\smallskip

	\textbf{Настройка имени пользователя и эл. почты на уровне системы}
	\begin{VerbatimCmd}
		
		> !color{orange}git config --global user.name "gavreg"
		> !color{orange}git config --global user.email "a.gavrilov@stankin.ru"
		
		> !color{orange}git config --global --list
		...
		user.email=a.gavrilov@stankin.ru
		user.name=gavreg
		...		
	\end{VerbatimCmd}
	
	\smallskip
	
	\begin{VerbatimCmd}
		
		> !color{orange}git config --global my.var "СТАНКИН"
		> !color{orange}git config --global --list
		...
		my.var=Станкин
		...
		
	\end{VerbatimCmd}
	
		
	
\end{frame}

				\begin{alertblock}{\Verb|git init|} \small
					команда инициализации репозитория. В текущей папке создаться скрытая папка .git, в которой будет размещён новый пустой репозиторий.

				3. Добавим файл ch1.txt в индекс.
				
				\begin{alertblock}{\Verb|git add| ---}
					команда Git, осуществляющая добавление файлов в индекс (stage).
				\end{alertblock}
				
				\begin{VerbatimCmd}[gobble=5,fontsize=\small]
					
					> !color{orange}git add ch1.txt 
					> !color{orange}git status
					
					Changes to be committed:
					  (use "git rm --cached <file>..." to unstage)
						!color{green}new file:   ch1.txt					
					
				\end{VerbatimCmd}
								
			\end{onlyenv}%			
			\begin{onlyenv}<4>%
				4. Применяем (фиксируем) изменения.
				
				\begin{alertblock}{Фиксация изменений (commit changes) --- }
				процедура переноса файлов из индекса в репозиторий. Перенос осуществляется хитро --- переносятся только отличия файлов индекса относительно тех, что есть в репозитории.		
				\end{alertblock}
				
				\begin{alertblock}{Коммит (commit) --- }
					содержит описание изменений файлов текущей версии проекта относительно предыдущей. Любой коммит, кроме самого первого, основан на более раннем (родительском).   		
				\end{alertblock}
				
			\end{onlyenv}%			
			\begin{onlyenv}<5>%
				4. Применяем (фиксируем) изменения.
					
				\begin{alertblock}{\Verb|git commit|}
					создаёт новый коммит, перенося в репозиторий то, что хранится в индексе в индексе. 
				\end{alertblock}
				
				\begin{VerbatimCmd}[gobble=5]

		\begin{alertblock}{Ветка (branch) ---}
		легковесный подвижный указатель на конкретный коммит в истории разработки. Она представляет собой изолированную линию разработки, позволяющую вести работу параллельно с другими линиями без влияния на них.
		\end{alertblock}	
		
		\begin{alertblock}{\Verb|git branch| ---}
			команда для управления ветками в Git. Позволяет создавать, просматривать, переименовывать и удалять ветки, но не переключается между ними.
		\end{alertblock}
		
		\begin{alertblock}{\Verb|git checkout| ---}
			git checkout — многофункциональная команда для: переключения между ветками, восстановления файлов, перемещения по истории коммитов					
		\end{alertblock}

		\end{alertblock}
		
		\begin{alertblock}{git merge --- }
			используется для того, чтобы взять изменения из одной или нескольких веток и интегрировать их в текущую активную ветку.

\begin{frame}\frametitle{Наименования веток}
	
	
	 
	\begin{columns}[]
		\edef\colA{0.35}
		\begin{column}[t]{\colA\linewidth}
			%Колнка1
			Как называть ветки:
			
			\begin{enumerate}
				\item Избегайте заглавных для единообразия.
				\item Не используйте пробелы.
				\item Название должно отражать суть:  \cancel{fix} fix-login-bug.
				\item Только буквы, цифры, дефисы, подчёркивания, слеши.
				\item Не начинайте с дефиса.
			\end{enumerate}
		\end{column}
		\begin{column}[t]{\fpeval{1-\colA}\linewidth}
			%Колнка2
			Типовые префиксы:
			\begin{itemize}
				\item feature/ ---	новая функциональность;	
				\item bugfix/ ---	исправление ошибки;	
				\item hotfix/ ---	критическое исправление в production;
				\item release/ ---	подготовка релиза;	
				\item refactor/	--- рефакторинг кода;	
				\item docs/ ---	обновление документации;	
				\item test/	--- добавление или изменение тестов;
				\item chore/ --- обслуживание проекта (зависимости, конфиги);	
				\item experiment/ --- Экспериментальные изменения;	
				\item wip/ --- work in progress — незавершённая работа;
			\end{itemize}
				
		\end{column}
	\end{columns}

		\Verb|git switch название_ветки|
		
		\item Создание и переключение на новую ветку:\\
		\Verb |git switch -c новая_ветка|\\
		(аналог \Verb|git checkout -b|)
		
		\item Переключение на предыдущую ветку:\\
		\Verb|git switch -|\\
		(возвращает на ветку, на которой вы были до этого)
		
		\item Принудительное переключение (отмена локальных изменений):\\
		\Verb|git switch -f ветка|\\
		(отбрасывает незакоммиченные изменения)
	\end{itemize}
	

	
	
\end{frame}
